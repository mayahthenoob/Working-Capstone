    // CLOCK LOGIC
    function updateClock() {
        const now = new Date();
        const dateEl = document.getElementById("date");
        const timeEl = document.getElementById("time");
        if(dateEl) dateEl.textContent = now.toDateString();
        if(timeEl) timeEl.textContent = now.toLocaleTimeString();
    }
    setInterval(updateClock, 1000);
    updateClock();

    // TAB LOGIC
    function openTab(evt, tabId) {
        const tabContents = document.getElementsByClassName("tab-content");
        for (let i = 0; i < tabContents.length; i++) {
            tabContents[i].classList.remove("active");
        }

        const tabBtns = document.getElementsByClassName("tab-btn");
        for (let i = 0; i < tabBtns.length; i++) {
            tabBtns[i].classList.remove("active");
        }

        document.getElementById(tabId).classList.add("active");
        evt.currentTarget.classList.add("active");
        
        if(tabId === 'class-tab') {
            refreshHeights();
        }
    }

    // COLLAPSE LOGIC

            const isExpanded = state[id] === undefined ? true : state[id];

            if (isExpanded) {
                wrapper.style.maxHeight = "0px";
                if(id === 'gen-wrapper' && btnId) document.getElementById(btnId).textContent = "Expand";
                if(arrow) arrow.style.transform = "rotate(-90deg)";
            } else {
                wrapper.style.maxHeight = wrapper.scrollHeight + "px";
                if(id === 'gen-wrapper' && btnId) document.getElementById(btnId).textContent = "Collapse";
                if(arrow) arrow.style.transform = "rotate(0deg)";
            }
            state[id] = !isExpanded;
        });
    }

    function applyFilters() {
        const matchType = document.getElementById('matchType').value;
        const searchField = document.getElementById('searchField').value;
        const keyword = filterInput.value.toLowerCase();
        
        const firstLetterActive = document.querySelector('#first-name-letters .active').getAttribute('data-letter');
        const lastLetterActive = document.querySelector('#last-name-letters .active').getAttribute('data-letter');

        const filtered = participantsData.filter(p => {
            const nameParts = p.name.split(' ');
            const firstName = nameParts[0] || "";
            const lastName = nameParts[1] || "";

            const matchesKeyword = keyword === "" || String(p[searchField] || "").toLowerCase().includes(keyword);
            const matchesFirst = firstLetterActive === 'all' || firstName.toUpperCase().startsWith(firstLetterActive);
            const matchesLast = lastLetterActive === 'all' || lastName.toUpperCase().startsWith(lastLetterActive);

            if (matchType === "all") {
                return matchesKeyword && matchesFirst && matchesLast;
            } else {
                const hasActiveFirstFilter = firstLetterActive !== 'all';
                const hasActiveLastFilter = lastLetterActive !== 'all';
                const hasKeywordFilter = keyword !== "";

                if (!hasActiveFirstFilter && !hasActiveLastFilter && !hasKeywordFilter) return true;

                return (hasKeywordFilter && matchesKeyword) || 
                       (hasActiveFirstFilter && matchesFirst) || 
                       (hasActiveLastFilter && matchesLast);
            }
        });

        renderParticipants(filtered);
    }
