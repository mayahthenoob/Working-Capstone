// ------------------------
// Words to Find Uppercase
// ------------------------
const amountInput = document.getElementById("amount");
const wordsContainer = document.getElementById("wordsContainer");

amountInput.addEventListener("input", function () {
    let amount = parseInt(this.value);
    if (amount > 12) { amount = 12; this.value = 12; }
    if (amount < 1 || isNaN(amount)) { wordsContainer.innerHTML = ""; return; }

    wordsContainer.innerHTML = "";
    for (let i = 1; i <= amount; i++) {
        const input = document.createElement("input");
        input.type = "text";
        input.placeholder = "Word " + i;
        input.required = true;

        input.addEventListener("input", () => {
            input.value = input.value.toUpperCase().replace(/[^A-Z]/g, '');
        });

        wordsContainer.appendChild(input);
    }
});

// ------------------------
// Description default/custom
// ------------------------
const defaultRadio = document.getElementById("default");
const customRadio = document.getElementById("custom");
const descriptionField = document.getElementById("description");

defaultRadio.addEventListener("change", function () {
    if (this.checked) {
        descriptionField.value = "This is a word search puzzle that should be completed in the allotted time";
        descriptionField.readOnly = true;
    }
});
customRadio.addEventListener("change", function () {
    if (this.checked) {
        descriptionField.value = "";
        descriptionField.readOnly = false;
    }
});

// ------------------------
// Format Opens/Closes
// ------------------------
function formatDateTime(value) {
    if (!value) return "";
    const date = new Date(value);
    return date.toLocaleString('en-GB', {
        weekday: 'long',
        day: 'numeric',
        month: 'long',
        year: 'numeric',
        hour: 'numeric',
        minute: '2-digit',
        hour12: true
    });
}

const opensInput = document.getElementById("opens");
const closesInput = document.getElementById("closes");

opensInput.addEventListener("change", function () {
    document.getElementById("opensFormatted").textContent = formatDateTime(this.value);
});

closesInput.addEventListener("change", function () {
    document.getElementById("closesFormatted").textContent = formatDateTime(this.value);
});

// ------------------------
// Time Limit display
// ------------------------
const timeLimitInput = document.getElementById("timeLimit");
const totalTimeDisplay = document.getElementById("totalTime");

timeLimitInput.addEventListener("input", function () {
    const minutes = parseInt(this.value);
    if (isNaN(minutes) || minutes < 0) { totalTimeDisplay.textContent = ""; return; }

    if (minutes < 60) {
        totalTimeDisplay.textContent = `${minutes} Minute${minutes !== 1 ? 's' : ''}`;
    } else {
        const hours = Math.floor(minutes / 60);
        const remainingMinutes = minutes % 60;
        if (remainingMinutes === 0) {
            totalTimeDisplay.textContent = `${hours} Hour${hours !== 1 ? 's' : ''}`;
        } else {
            totalTimeDisplay.textContent =
                `${hours} Hour${hours !== 1 ? 's' : ''} : ${remainingMinutes} Minute${remainingMinutes !== 1 ? 's' : ''}`;
        }
    }
});

// ------------------------
// Word Search Design Grid
// ------------------------
const designGrid = document.getElementById("designGrid");
const gridSize = 10;
let designMode = "custom";

// Helper: get target cell
function focusCell(r, c) {
    const target = designGrid.querySelector(`.cell[data-row='${r}'][data-col='${c}']`);
    if (target) {
        target.focus();
        // Place cursor at the end
        const range = document.createRange();
        const sel = window.getSelection();
        range.selectNodeContents(target);
        range.collapse(false);
        sel.removeAllRanges();
        sel.addRange(range);
    }
}

// Create grid
for (let r = 0; r < gridSize; r++) {
    for (let c = 0; c < gridSize; c++) {
        const cell = document.createElement("div");
        cell.className = "cell";
        cell.dataset.row = r;
        cell.dataset.col = c;
        cell.contentEditable = true;
        cell.style.textAlign = "center";

        // Input handler: auto-uppercase and advance
        cell.addEventListener("input", (e) => {
            let char = cell.textContent.toUpperCase().replace(/[^A-Z]/g, '');
            char = char.charAt(0) || "";
            cell.textContent = char;

            // Maintain cursor position
            const range = document.createRange();
            const sel = window.getSelection();
            range.selectNodeContents(cell);
            range.collapse(false);
            sel.removeAllRanges();
            sel.addRange(range);

            if (char.length > 0) {
                if (c < gridSize - 1) focusCell(r, c + 1);
                else if (r < gridSize - 1) focusCell(r + 1, 0);
            }
        });

        // Keydown handler: Arrow Navigation & Hold-to-Delete Backspace
        cell.addEventListener("keydown", (e) => {
            switch (e.key) {
                case "ArrowUp":
                    e.preventDefault();
                    focusCell(Math.max(0, r - 1), c);
                    break;
                case "ArrowDown":
                    e.preventDefault();
                    focusCell(Math.min(gridSize - 1, r + 1), c);
                    break;
                case "ArrowLeft":
                    e.preventDefault();
                    focusCell(r, Math.max(0, c - 1));
                    break;
                case "ArrowRight":
                    e.preventDefault();
                    focusCell(r, Math.min(gridSize - 1, c + 1));
                    break;
                case "Backspace":
                    // If current cell has text, clear it
                    if (cell.textContent.length > 0) {
                        cell.textContent = "";
                        // e.preventDefault() not strictly needed here as we want the backspace action
                    } else {
                        // If cell is already empty, move focus back and clear that cell too
                        e.preventDefault();
                        let prevR = r;
                        let prevC = c - 1;

                        if (prevC < 0) {
                            if (r > 0) {
                                prevR = r - 1;
                                prevC = gridSize - 1;
                            } else {
                                return; // Top of grid
                            }
                        }

                        const prevCell = designGrid.querySelector(`.cell[data-row='${prevR}'][data-col='${prevC}']`);
                        if (prevCell) {
                            prevCell.textContent = "";
                            focusCell(prevR, prevC);
                        }
                    }
                    break;
                default:
                    if (e.key.length === 1 && !e.key.match(/[a-zA-Z]/)) {
                        e.preventDefault();
                    }
            }
        });

        // Paste support
        cell.addEventListener("paste", (e) => {
            e.preventDefault();
            let pasteText = (e.clipboardData || window.clipboardData).getData('text');
            pasteText = pasteText.toUpperCase().replace(/[^A-Z]/g, '');
            let currR = r, currC = c;

            for (let char of pasteText) {
                const target = designGrid.querySelector(`.cell[data-row='${currR}'][data-col='${currC}']`);
                if (!target) break;
                target.textContent = char;
                if (currC < gridSize - 1) currC++;
                else if (currR < gridSize - 1) { currC = 0; currR++; }
                else break;
            }
            focusCell(currR, currC);
        });

        designGrid.appendChild(cell);
    }
}

// Mode radio buttons
const modeCustomRadio = document.getElementById("modeCustom");
const modeDefaultRadio = document.getElementById("modeDefault");

modeDefaultRadio.addEventListener("change", () => {
    if (modeDefaultRadio.checked) {
        designMode = "default";
        const letters = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
        designGrid.querySelectorAll(".cell").forEach(cell => {
            if (!cell.textContent.trim()) {
                cell.textContent = letters.charAt(Math.floor(Math.random() * letters.length));
            }
        });
    }
});

modeCustomRadio.addEventListener("change", () => { if (modeCustomRadio.checked) designMode = "custom"; });

// Publish data
document.querySelector(".publish-btn").addEventListener("click", () => {
    const gridData = [];
    for (let r = 0; r < gridSize; r++) {
        const row = [];
        for (let c = 0; c < gridSize; c++) {
            row.push(designGrid.querySelector(`.cell[data-row='${r}'][data-col='${c}']`).textContent || "");
        }
        gridData.push(row);
    }
    console.log("Grid Data:", gridData);
    alert("Data saved to console!");
});

<script>
  const countdownDate = new Date().getTime() + (20 * 60 * 1000); 

  function updateCountdown() {
    const now = new Date().getTime();
    let timeLeft = countdownDate - now;

    if (timeLeft < 0) {
      clearInterval(interval);
      document.querySelector(".timer-container").innerHTML = "<h1 style='color:#222'>Time's Up!</h1>";
      return;
    }

    const h = Math.floor((timeLeft / (1000 * 60 * 60)) % 24);
    const m = Math.floor((timeLeft % (1000 * 60 * 60)) / (1000 * 60));
    const s = Math.floor((timeLeft % (1000 * 60)) / 1000);

    document.getElementById("hours").innerText = h.toString().padStart(2,'0');
    document.getElementById("minutes").innerText = m.toString().padStart(2,'0');
    document.getElementById("seconds").innerText = s.toString().padStart(2,'0');
  }

  const interval = setInterval(updateCountdown, 1000);
  updateCountdown();

  // Word Search Game Logic
  const gridData = [
    "ANGRYSAVRN","XRPETKYENS","RDETICXEEU","FSVDANGERR",
    "FMCCAUASVP","CWAACMNAOR","AWORRIEDUI","LIMEMEEOSS",
    "MTIREDDBIE","SADHAPPYXD"
  ];
  const words = ["ANGRY","HAPPY","SCARED","WORRIED","SURPRISED","NERVOUS","CALM","TIRED","SAD","EXCITED","TED","DANGER"];
  const colors = [
    "var(--curious-blue)","var(--blush)","var(--san-juan)","var(--tiber)",
    "var(--cherry-red)","var(--mulled-wine)","var(--pesto)","var(--sorbet-orange)",
    "var(--forest-green)","var(--soft-jade)","var(--radical-red)"
  ];

  let colorIndex = 0, isSelecting=false, startCell=null, currentSelection={cells:[],angle:0,width:0,left:0,top:0}, wordColors={}, cellsArray=[];
  const grid=document.getElementById("grid"), wordList=document.getElementById("wordList"), tempPill=document.getElementById("temp-pill");

  function buildGrid() {
    gridData.forEach((row,r)=>{
      row.split("").forEach((letter,c)=>{
        const cell=document.createElement("div");
        cell.className="cell";
        cell.textContent=letter;
        cell.dataset.row=r;
        cell.dataset.col=c;
        cell.addEventListener("mousedown",(e)=>{e.preventDefault(); handleStart(cell);});
        cell.addEventListener("mouseenter",()=>handleDrag(cell));
        grid.appendChild(cell);
        cellsArray.push(cell);
      });
    });
  }

  function buildWords() {
    words.forEach(word=>{
      const li=document.createElement("li");
      li.textContent=word;
      li.id=word;
      wordList.appendChild(li);
    });
  }

  function handleStart(cell){
    isSelecting=true;
    startCell=cell;
    const color=colors[colorIndex % colors.length];
    document.documentElement.style.setProperty("--current-color", color);
    tempPill.style.display="block";
    updateSelection(cell);
  }

  function handleDrag(cell){
    if(!isSelecting) return;
    updateSelection(cell);
  }

  document.addEventListener("mouseup",()=>{
    if(!isSelecting) return;
    isSelecting=false;
    checkSelection();
    tempPill.style.display="none";
  });

  function updateSelection(endCell){
    const r1=parseInt(startCell.dataset.row), c1=parseInt(startCell.dataset.col);
    const r2=parseInt(endCell.dataset.row), c2=parseInt(endCell.dataset.col);
    const dr=r2-r1, dc=c2-c1;
    const absDr=Math.abs(dr), absDc=Math.abs(dc);
    if(!(dr===0||dc===0||absDr===absDc)) return;

    const cellSize=50;
    const steps=Math.max(absDr,absDc);
    const actualDist=Math.sqrt(dr*dr+dc*dc)*cellSize;
    const pillWidth=actualDist+40;
    const angle=Math.atan2(dr,dc)*(180/Math.PI);
    const startX=c1*cellSize+(cellSize/2), startY=r1*cellSize+(cellSize/2);

    tempPill.style.width=`${pillWidth}px`;
    tempPill.style.left=`${startX-20}px`;
    tempPill.style.top=`${startY}px`;
    tempPill.style.transform=`rotate(${angle}deg)`;

    currentSelection.width=pillWidth;
    currentSelection.angle=angle;
    currentSelection.left=tempPill.style.left;
    currentSelection.top=tempPill.style.top;
    currentSelection.cells=[];

    const rStep=dr===0?0:dr/absDr, cStep=dc===0?0:dc/absDc;
    for(let i=0;i<=steps;i++){
      const r=r1+i*rStep, c=c1+i*cStep;
      currentSelection.cells.push(cellsArray[r*10+c]);
    }
  }

  function checkSelection(){
    const word=currentSelection.cells.map(c=>c.textContent).join("");
    const reversed=word.split("").reverse().join("");
    const match=words.find(w=>(w===word||w===reversed)&&!wordColors[w]);
    if(match){
      const color=colors[colorIndex % colors.length];
      wordColors[match]=color;
      const pill=document.createElement("div");
      pill.className="pill-highlight";
      pill.style.setProperty("--pill-color", color);
      pill.style.width=`${currentSelection.width}px`;
      pill.style.left=currentSelection.left;
      pill.style.top=currentSelection.top;
      pill.style.transform=`rotate(${currentSelection.angle}deg)`;
      grid.appendChild(pill);

      const li=document.getElementById(match);
      li.classList.add("found");
      li.style.color=color;
      colorIndex++;

      // Show modal if all words found
      if(words.every(w => wordColors[w])) {
        document.getElementById('finishModal').style.display='block';
      }
    }
  }

  // Modal buttons
  document.getElementById('continueBtn').addEventListener('click', () => {
      document.getElementById('finishModal').style.display = 'none';
  });

  document.getElementById('submitBtn').addEventListener('click', () => {
      alert("Quiz submitted!"); // Replace with real submit logic
      document.getElementById('finishModal').style.display = 'none';
  });

  buildGrid();
  buildWords();
</script>
